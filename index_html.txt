const text =
`Dear Ridhima,

Happy Birthday! 🎉

Today is all about celebrating the wonderful person you are.

Thank you for being the kind, caring, and genuine person that you are. Your smile, your positivity, and your presence make a difference in ways you may not even realize.

Happy Birthday, Ridhima! ❤️✨

function showWish(){
let i=0;
const msg=document.getElementById("message");
msg.innerHTML="";

const typing=setInterval(()=>{
msg.innerHTML+=text.charAt(i);
i++;

if(i>=text.length){
clearInterval(typing);
}
},25);
}

function createHeart(){
const heart=document.createElement("div");
heart.classList.add("heart");
heart.innerHTML="❤️";
heart.style.left=Math.random()*100+"vw";
heart.style.fontSize=(Math.random()*20+15)+"px";
heart.style.animationDuration=(Math.random()*3+4)+"s";

document.body.appendChild(heart);

setTimeout(()=>{
heart.remove();
},7000);
}

setInterval(createHeart,300);
